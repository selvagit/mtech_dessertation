@startuml
title ecpri 

state  ecpri {
    state reset

state ecpri_rx {
  state reset
  state cpri_hdr
  state read
  state write
  reset_rx --> cpri_hdr : found cpri header
  cpri_hdr --> read : read flg set
  cpri_hdr --> write : write flg set 
  write --> write_to_mem : valid memory address
  write_to_mem --> reset_rx : mem write successfull/create_tx_pkt, create_pkt 
  cpri_hdr --> reset_rx : invalid flag
  read -->reset_rx : invalid payload length
  write -->reset_rx : invalid payload length
  write_to_mem --> reset_rx : invalid mem address
}

state ecpri_tx {
  state reset_tx
  state tx_resp
  state  rx_resp
  reset_tx-->eth_hdr : create_pkt 
  eth_hdr-->ip_hdr : create ip header
  ip_hdr --> udp_hdr : create udp header 
  udp_hdr-->rx_resp : create_rx_pkt
  udp_hdr-->tx_resp : create_tx_pkt
  tx_resp-->complete : send_packet
  rx_resp-->copy_from_mem: valid memory
  rx_resp-->complete : invalid memory
  copy_from_mem --> complete : copy complete
  complete -> reset_tx : reset_ram
}

reset--> ecpri_tx : create_pkt
reset--> ecpri_rx : create_rx_pkt

ecpri_rx -Right--> ecpri_tx : create_pkt
ecpri_rx -Right--> ecpri_tx : create_rx_pkt
ecpri_rx -Right--> ecpri_tx : create_tx_pkt 

}
@enduml
